extern printf
section .data
message: db "%i",10,0
v: db 0
b: db 0
a: db 0

section .text
global main

main :
	push eax
	mov eax,5
	mov [v], eax
	push ebx
	mov ebx,3
	ADD ebx,[v]	mov [b], ebx
	push ecx
	mov ecx,2
	SUB [b],ecx
	push dword [b]
	push dword message
	call printf
	add esp,8
	mov ecx,2
	mov [a], ecx
	ret
